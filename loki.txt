    <script>
        // Dark/Light mode toggle
        const themeSwitch = document.getElementById('themeSwitch');
        const body = document.body;
        
        themeSwitch.addEventListener('click', () => {
            body.classList.toggle('light-mode');
            themeSwitch.innerHTML = body.classList.contains('light-mode') ? 
                '<i class="fas fa-sun fa-2x"></i>' : 
                '<i class="fas fa-moon fa-2x"></i>';
        });

        // Scroll progress indicator
        window.onscroll = () => {
            const winScroll = document.body.scrollTop || document.documentElement.scrollTop;
            const height = document.documentElement.scrollHeight - document.documentElement.clientHeight;
            const scrolled = (winScroll / height) * 100;
            document.getElementById('progressBar').style.width = scrolled + "%";
        };

        // Typewriter effect for subtitle
        const subtitle = document.getElementById('subtitle');
        const textArray = ["God of Mischief & Master Developer", "Chaos Engineer", "Shapeshifting Specialist"];
        let index = 0;

        function typeWriter(text, i = 0) {
            if (i < text.length) {
                subtitle.innerHTML = text.substring(0, i+1);
                setTimeout(() => typeWriter(text, i+1), 100);
            } else {
                setTimeout(() => deleteText(text), 2000);
            }
        }

        function deleteText(text) {
            if (subtitle.innerHTML.length > 0) {
                subtitle.innerHTML = subtitle.innerHTML.slice(0, -1);
                setTimeout(() => deleteText(text), 50);
            } else {
                index = (index + 1) % textArray.length;
                typeWriter(textArray[index]);
            }
        }

        typeWriter(textArray[index]);

        // Dynamic content loading
        const projects = [
            { 
                title: "Chaos Engine", 
                tags: ["Python", "AI", "Flask"],
                description: "Advanced machine learning platform for chaotic pattern analysis",
                year: 2023
            },

                title: "Dimension Portal", 
                tags: ["JavaScript", "Three.js", "WebGL"],
                description: "3D visualization tool for multidimensional data exploration",
                year: 2022
            }
        ];

        const certifications = [

        // Render projects
        const projectsContainer = document.getElementById('projectsContainer');
        projects.forEach(project => {
            const projectCard = document.createElement('div');
            projectCard.className = 'project-card';
            projectCard.innerHTML = `
                <h3>${project.title} (${project.year})</h3>
                <div class="tags">${project.tags.map(tag => `<span class="tag">${tag}</span>`).join('')}</div>
                <p>${project.description}</p>
            `;
            projectsContainer.appendChild(projectCard);
        });

        // Render certifications
        const certificationsContainer = document.getElementById('certificationsContainer');
        certifications.forEach(cert => {
            const certCard = document.createElement('div');
            certCard.className = 'certification-card';
            certCard.innerHTML = `
                <h3>${cert.title}</h3>
                <p>${cert.issuer} - ${cert.year}</p>
                <p class="credential">Credential ID: ${cert.credential}</p>
            `;
            certificationsContainer.appendChild(certCard);
        });

        // Parallax effect on header
        window.addEventListener('scroll', () => {
            const header = document.querySelector('header');
            const scrolled = window.pageYOffset;
            header.style.backgroundPositionY = scrolled * 0.5 + 'px';
